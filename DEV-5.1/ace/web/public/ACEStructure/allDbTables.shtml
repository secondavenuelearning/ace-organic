Primary keys

To list primary keys:

select substr(cons.constraint_name, 1, 25) as constraint_name,
substr(cols.table_name || '.' || cols.column_name, 1, 45) as key
from user_constraints cons, user_cons_columns cols
where cons.constraint_name = cols.constraint_name
and cons.constraint_type = 'P' 
and cons.constraint_name like 'P_%'
order by cols.table_name, cols.column_name;

Foreign keys

To list foreign keys:

select con.constraint_name,
	con.table_name || '.' || con.column_name || ' -> ' ||
		rel.table_name || '.' || rel.column_name as key
from user_cons_columns con 
join user_constraints cc 
	on con.constraint_name = cc.constraint_name
join user_cons_columns rel 
	on cc.r_constraint_name = rel.constraint_name 
	and con.position = rel.position
where cc.constraint_type = 'R'
order by con.constraint_name;
